üìù Send Text Message

curl -X POST https://whatsapp.giftedtech.co.ke/api/sendMessage.php \
  -H "Content-Type: application/json" \
  -d '{
    "number": "2547XXXXXXXX",
    "type": "text",
    "message": "Hello from WaSender API!"
  }'
const response = await fetch('https://whatsapp.giftedtech.co.ke/api/sendMessage.php', {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({
    number: '2547XXXXXXXX',
    type: 'text',
    message: 'Hello from WaSender API!'
  })
});

const data = await response.json();
console.log(data);
import requests

response = requests.post(
    'https://whatsapp.giftedtech.co.ke/api/sendMessage.php',
    json={
        'number': '2547XXXXXXXX',
        'type': 'text',
        'message': 'Hello from WaSender API!'
    }
)

print(response.json())
